    return np.sum((y_data - y_model)**2.)/sigma**2.

def poly_mle(p, t, y, e):
    #Formamos un sistema Ax=b
    A = np.empty([p+1,p+1])
    b = np.empty(p+1)
    
    for i in xrange(len(A)):
        for j in xrange(len(A)):
            A[i,j] = np.sum(t**(i+j)/e**2)
    
    for i in xrange(len(b)):
        b[i] = np.sum(y*t**i/e**2)
    
    return np.linalg.solve(A,b)

def chi_sq_cuadratico(a_0_model, a_1_model, a_2_model, x_data, y_data, sigma):
    y_model = a_0_model + a_1_model*x_data + a_2_model*x_data**2.
    return np.sum((y_data - y_model)**2.)

def chi_sq_cubico(a_0_model, a_1_model, a_2_model, a_3_model, x_data, y_data, sigma):
    y_model = a_0_model + a_1_model*x_data + a_2_model*x_data**2. + a_3_model*x_data**3.
    return np.sum((y_data - y_model)**2.)

def chi_sq_cuartico(a_0_model, a_1_model, a_2_model,  a_3_model, a_4_model, x_data, y_data, sigma):
    y_model = a_0_model + a_1_model*x_data + a_2_model*x_data**2. + a_3_model*x_data**3. + a_4_model*x_data**4.
    return np.sum((y_data - y_model)**2.)

def chi_sq_quintico(a_0_model, a_1_model, a_2_model,  a_3_model, a_4_model, a_5_model, x_data, y_data, sigma):
    y_model = a_0_model + a_1_model*x_data + a_2_model*x_data**2. + a_3_model*x_data**3. + a_4_model*x_data**4. + a_5_model*x_data**5.
    return np.sum((y_data - y_model)**2.)
