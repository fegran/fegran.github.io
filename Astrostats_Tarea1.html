plt.figure(figsize=(6,6))
plt.subplots_adjust(hspace=0.001, wspace=0.001)

repeticiones = 20000
size=2000.

A = np.array([])
B = np.array([])
AB = np.array([])
P_A = 1/2.
P_B = 2/3.
P_AB = 1/3.

for i in xrange(repeticiones):
    evento = np.random.randint(1, 7, size=size)
    unique, counts = np.unique(evento, return_counts=True)
    A = np.append(A, (counts[unique==2] + counts[unique==4] + counts[unique==6])/size)
    B = np.append(B, (counts[unique==1] + counts[unique==2] + counts[unique==3] + counts[unique==4])/size)
    AB = np.append(AB, (counts[unique==2] + counts[unique==4])/size)

plt.subplot(223)    
plt.plot(np.linspace(0,1),np.linspace(0,1), color='g', alpha=0.8, linestyle='dashed')
plt.scatter(A*B, AB, s=3, alpha=0.4, color='grey')
plt.scatter(P_A*P_B, P_AB, marker='*', color='r', s=100)
plt.xlabel(r'$P(A)P(B)$', size=17)
plt.ylabel(r'$P(AB)$', size=17)
plt.ylim(0.28, 0.38)
plt.xlim(0.28, 0.38)

plt.subplot(221)
plt.hist(A*B, bins=75, histtype='stepfilled', lw=2.5, color='grey', alpha=0.4)
plt.axvline(P_A*P_B, 0, 1000, color='red', lw=2, alpha=0.5)
plt.tick_params(axis='x', labelbottom='off')
plt.tick_params(axis='y', labelleft='off')
plt.xlim(0.28, 0.38)

plt.subplot(224)
plt.hist(AB, bins=75, histtype='stepfilled', lw=2.5, color='grey', alpha=0.4, orientation=u'horizontal')
plt.axhline(P_AB, 0, 1000, color='red', lw=2, alpha=0.5)
plt.tick_params(axis='y', labelleft='off')
plt.tick_params(axis='x', labelbottom='off')
plt.ylim(0.28, 0.38)

plt.savefig('dice_indep.pdf', format='pdf')
plt.show()
