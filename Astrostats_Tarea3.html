def poly_mle(p, t, y, e):
    #Formamos un sistema Ax=b
    A = np.empty([p+1,p+1])
    b = np.empty(p+1)
    
    for i in xrange(len(A)):
        for j in xrange(len(A)):
            A[i,j] = np.sum(t**(i+j)/e**2)
    
    for i in xrange(len(b)):
        b[i] = np.sum(y*t**i/e**2)
    
    return np.linalg.solve(A,b)

def solution(x, t):
    t = np.linspace(t.min(), t.max(), len(t))
    s = 0
    for i, x_i in enumerate(x):
        s = s + x_i*t**i
    return s

def chisq(p, t, y, e):
    x = poly_mle(p, t, y, e)
    
    chi = 0
    for i in xrange(len(t)):
        poly_sum = 0
        for j in range(p):
            poly_sum = poly_sum + x[j]*t[i]**j
        chi = chi + 1/e[i]**2.*(y[i] - poly_sum)**2.
    return chi

def var_a0(t, y, sigma):
    N = len(t)      
    return (sigma**2*np.sum(t**2))/(N*np.sum(t**2)-np.sum(t)**2)

def var_a1(t, y, sigma):
    N = len(t)
    return N*sigma**2/(N*np.sum(t**2)-np.sum(t)**2)
