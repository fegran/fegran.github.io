#Galactic coordinates of the RRab candidates and BEAM extinction map

def HMS(hours, pos):
    minutes = np.abs ((hours-int(hours) ) * 60.) 
    seconds = 0
    hours = int(hours)
    if hours == 0:
        if minutes == 0:
            if seconds == 0:
                return r"$%d^\circ%02d'$" % (hours, minutes)
            return "%d''" % (seconds)
        return "%d'%02d''" % (minutes, seconds)
    return r"$%d^\circ%02d'$" % (hours, minutes)

RA, DEC = np.genfromtxt('lb.dat', usecols=(1,2), unpack=True)
l_neg, b_neg, AKs_neg = np.genfromtxt('beamn.dat', usecols=(0,1,5), unpack=True)
l_pos, b_pos, AKs_pos = np.genfromtxt('beamp.dat', usecols=(0,1,5), unpack=True)

plt.figure(figsize=(16,5))
plt.gca().xaxis.set_major_formatter(plt.FuncFormatter(HMS))
plt.gca().yaxis.set_major_formatter(plt.FuncFormatter(HMS))

RRL = SkyCoord(ra=RA, dec=DEC ,unit=(u.degree, u.degree), frame='icrs')
gal = RRL.galactic
l = np.array([])
for i,ell in enumerate(gal.l.deg):
    if ell > 10.:
        l = np.append(l, ell - 360.)
    else:
        l = np.append(l, ell)
b = gal.b.deg

neg = plt.hexbin(l_neg, b_neg, AKs_neg, gridsize=80, cmap=plt.get_cmap('gist_yarg'), vmax=0.20)
pos = plt.hexbin(l_pos, b_pos, AKs_pos, gridsize=80, cmap=plt.get_cmap('gist_yarg'), vmax=0.20)
plt.scatter(l, b, marker='*', color='r', label='RRab', s=50.)

cb = plt.colorbar(pos, pad=0.01, shrink=1.0)
cb.set_label(r'$\langle A_{K_s} \rangle$', fontsize=19)

plt.xlabel(r'$\ell$', fontsize=19)
plt.ylabel('$b$', fontsize=19)

plt.xlim(10.1,-10.1)
plt.ylim(-10.35,-7.9)
plt.savefig('lb.pdf', format='pdf')
